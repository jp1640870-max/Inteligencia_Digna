CREATE TABLE IF NOT EXISTS hearts (
  id              TEXT PRIMARY KEY,
  user_id         TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  name            TEXT NOT NULL,
  role            TEXT DEFAULT '',          -- Mentor, Coach, Analista, etc.
  tone            TEXT DEFAULT '',          -- Formal, Casual, Técnico, etc.
  instructions    TEXT DEFAULT '',          -- Instrucciones detalladas
  limitations     TEXT DEFAULT '',          -- Lo que NO debe hacer
  temperature     REAL DEFAULT 0.7,
  knowledge_files TEXT DEFAULT '[]',        -- JSON array [{name, content, size}]
  tools           TEXT DEFAULT '[]',        -- JSON array ["read_files", "generate_docs"]
  agent_memory    TEXT DEFAULT '',          -- Memoria persistente entre sesiones
  is_public       INTEGER DEFAULT 0,        -- 1 = compartido con la empresa
  is_preset       INTEGER DEFAULT 0,        -- 1 = heart precargado por admin
  created_at      TEXT DEFAULT (datetime('now')),
  updated_at      TEXT DEFAULT (datetime('now'))
);

CREATE INDEX IF NOT EXISTS idx_hearts_user ON hearts(user_id);
CREATE INDEX IF NOT EXISTS idx_hearts_public ON hearts(is_public);

export function buildMessages(
  history: HistoryItem[],
  newMessage?: string,
  filesContent?: string,
  projectContext?: string,
  heart?: Heart | null
): OllamaMessage[] {
  const parts: string[] = [
    "Responde en español de forma clara, directa y útil.",
    "Si das código, usa bloques markdown con el lenguaje correspondiente.",
    "NO uses HTML.",
    "No repitas la pregunta.",
    "Sé conciso pero completo.",
  ];

  // ─── HEART: personalidad ───
  if (heart) {
    if (heart.role) {
      parts.push(`\nEres un ${heart.role}.`);
    }
    if (heart.tone) {
      parts.push(`\nTONO: ${heart.tone}.`);
    }
    if (heart.instructions) {
      parts.push(`\nINSTRUCCIONES:\n${heart.instructions}`);
    }
    if (heart.limitations) {
      parts.push(`\nLIMITACIONES:\n${heart.limitations}`);
    }
    if (heart.tools?.length) {
      parts.push(`\nHERRAMIENTAS DISPONIBLES:\n${
        heart.tools.map(t => `- ${getToolDescription(t)}`).join('\n')
      }`);
    }
    if (heart.agent_memory) {
      parts.push(`\nNOTAS PERSISTENTES:\n${heart.agent_memory}`);
    }
    if (heart.knowledge_files?.length) {
      parts.push(`\nINFORMACIÓN DE REFERENCIA:\n${
        heart.knowledge_files.map(f => f.content).join('\n---\n')
      }`);
    }
  }

  // ─── PROYECTO ───
  if (projectContext) {
    parts.push(`\nCONTEXTO DEL PROYECTO:\n${projectContext}`);
  }

  const systemPrompt = parts.join('\n');
  // ...resto igual, construye messages con history + newMessage
}

function getToolDescription(tool: string): string {
  const map: Record<string, string> = {
    read_files: "Lee y analiza archivos PDF, DOCX, XLSX, TXT subidos por el usuario",
    generate_docs: "Genera documentos Word/Excel con los datos de la conversación",
    web_search: "Busca información actualizada en internet",
    analyze_images: "Analiza imágenes y extrae información visual",
  };
  return map[tool] || tool;
}

Cuando la conversación supera ~12K tokens, el servidor:
1. Pide a Ollama: "Resume esta conversación en 2 párrafos"
2. Guarda el resumen como un mensaje system en la DB
3. En la siguiente request, envía: [RESUMEN] + [últimos 5 mensajes]

Persistencia:
Cada agente tiene un campo agent_memory (texto) que persiste entre sesiones. El agente puede actualizarlo con instrucciones como:
"Actualiza tu nota: El cliente prefiere contacto por email"
Al inicio de cada conversación se inyecta:
NOTAS PERSISTENTES:
[agent_memory]
Así el agente recuerda información clave aunque pasen días entre usos.

Aplicable en chat general? learning? me estoy volviendo loco?

Diagrama de flujo de contexto
SIN MEMORIA:
  [System Prompt] ── [Historial completo ~8 msg] ── [Nuevo mensaje]
                    ↑ Se pierde cuando llega a ~16K

CON RESUMEN:
  [System Prompt] ── [RESUMEN] ── [Últimos 5 msg] ── [Nuevo mensaje]
                    ↑ Ocupa ~500 tokens, no crece

CON MEMORIA PERSISTENTE:
  [System Prompt] ── [NOTAS] ── [RESUMEN] ── [Últimos 5 msg] ── [Nuevo mensaje]
                    ↑ ~300 tokens        ↑ ~500 tokens

Implementación de herramientas:
Tabla agents, columna tools (JSON):
["read_files", "generate_docs", "search_knowledge"]
Cuando el usuario envía un mensaje, el server:
1. Detecta qué herramientas tiene el agente
2. Las inyecta en el system prompt como capacidades disponibles
3. El modelo "decide" usarlas y el backend las ejecuta
Ejemplo en el prompt:
HERRAMIENTAS DISPONIBLES:
- read_files: Lee y analiza PDF, DOCX, XLSX subidos por el usuario
- generate_docs: Genera documentos Word/Excel con los datos de la conversación
Para generate_docs, el flujo sería:
1. El modelo responde: [GENERAR_DOC:reporte_ventas.docx]
2. El frontend/backend intercepta, llama a lib/doc-generator.ts
3. Devuelve el archivo generado al usuario
